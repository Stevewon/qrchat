rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // 기존 규칙 유지
    match /{document=**} {
      allow read: if true;
      allow write: if request.auth != null;
    }
    
    // sticker_packs 컬렉션만 웹에서도 쓰기 가능하도록 임시 허용
    match /sticker_packs/{packId} {
      allow read, write: if true;  // ⚠️ 개발/테스트 전용
    }
  }
}
