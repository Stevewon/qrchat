rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    // 기존 규칙 유지
    match /{allPaths=**} {
      allow read: if true;
      allow write: if request.auth != null;
    }
    
    // 스티커 폴더만 웹에서도 업로드 가능하도록 임시 허용
    match /stickers/{allPaths=**} {
      allow read, write: if true;  // ⚠️ 개발/테스트 전용
    }
  }
}
